Example: automate vulnerability evidence

Schedule a daily job that pushes the latest scan summary into Complyo and links it to your vulnerability management control.

import Complyo from '@complyo/sdk';

const client = new Complyo({
  clientId: process.env.COMPLYO_CLIENT_ID,
  clientSecret: process.env.COMPLYO_CLIENT_SECRET,
});

const evidence = await client.evidence.create({
  controlId: 'ctrl_vuln_monitoring',
  title: 'Daily vulnerability scan',
  collectedAt: new Date().toISOString(),
  metadata: {
    scanner: 'Orca Security',
    findings: 0,
    reportUrl: 'https://scanner/reports/123',
  },
});

console.log('Evidence stored: ' + evidence.id);
